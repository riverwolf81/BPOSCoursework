start_file=start.elf
fixup_file=fixup.dat

kernel=Image

# To use an external initramfs file
#initramfs rootfs.cpio.gz

# Disable overscan assuming the display supports displaying the full resolution
# If the text shown on the screen disappears off the edge, comment this out
disable_overscan=1

# How much memory in MB to assign to the GPU on Pi models having
# 256, 512 or 1024 MB total memory
gpu_mem_256=100
gpu_mem_512=100
gpu_mem_1024=100

# Enable VC6 GPU (for Raspberry Pi 5)
gpu_mem=128              # Allocate 128MB of RAM to the GPU
dtoverlay=vc6-kms-v3d    # Enable VC6 Kernel Mode Setting (KMS) for Raspberry Pi 5

# Enable HDMI output (if using HDMI display)
hdmi_force_hotplug=1     # Force HDMI output even if no display is detected
hdmi_group=1             # HDMI group 1 (CEA, for TVs)
hdmi_mode=16             # HDMI mode 16 (1080p)

# Set framebuffer resolution
framebuffer_width=1920   # Set framebuffer width to 1920 pixels
framebuffer_height=1080  # Set framebuffer height to 1080 pixels

# Enable UART0 for serial console on ttyAMA0
dtoverlay=miniuart-bt
